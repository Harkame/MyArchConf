loadkeys fr-pc
cfdisk /dev/sda
		swap - <RAM>
		/    - <RESTE> - bootable
		
mkswap /dev/sda1
swapon /dev/sda1

mkfs.btrfs -L "Arch" -f -n 65536 /dev/sda2

mount /dev/sda2 /mnt

nano /etc/pacman.d/mirrorlist
	ctrl^w
	ctrl^r
	Server -> #Server
	
//Sauf le premier

pacstrap /mnt base base-devel networkmanager grub
genfstab -p -U /mnt >> /mnt/etc/fstab

arch-chroot /mnt
	systemctl enable networkmanager.service
	echo "KEYMAP=fr-latin9" >> /etc/vconsole.conf
	echo "FONT=lat9w-16" >> /etc/vconsole.conf

	nano /etc/locale.gen
		#fr_FR.UTF-8 -> fr_FR.UTF-8

	nano /etc/pacman.conf
		+[archlinuxfr]
		+SigLevel = Never
		+Server = http://repo.archlinux.fr/$arch
	
		+[blackarch]
		+SigLevel = Never
		+Server = http://blackarch.org/blackarch/$repo/os/$arch

	
	locale-gen
	echo "LANG=fr_FR.UTF-8" >> /etc/locale.conf
	echo "LC_COLLATE=C" >> /etc/locale.conf
	export LANG=fr_FR.UTF-8

	ln -sf /usr/share/zoneinfo/Europe/Paris /etc/localtime
	hwclock --systohc --utc

	echo "Skadi" >> /etc/hostname

	nano /etc/mkinitcpio.conf
		fsck -> btrfs
	
	passwd root
		~
		~

exit

grub-mkconfig -o /boot/grub/grub.cfg
grub-install --no-floppy --recheck /dev/sda

post-install

pacman -S xorg-server xorg-xinit xorg-xmessage xorg-utils rxvt-unicode
